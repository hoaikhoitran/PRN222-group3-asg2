1) Delegate callback cơ bản (không dùng event)
using System;

public delegate double MathOperation(double a, double b);

public class Calculator
{
    public double Compute(double x, double y, MathOperation op)
    {
        return op(x, y);
    }
}

2) Multicast delegate (+=, -=)
using System;

public delegate void NotifyDelegate(string message);

public class Notifier
{
    public NotifyDelegate OnNotify; // delegate field (không phải event)

    public void Raise(string msg)
    {
        OnNotify?.Invoke(msg);
    }
}

public class LogA
{
    public void WriteA(string msg) => Console.WriteLine($"A: {msg}");
}

public class LogB
{
    public void WriteB(string msg) => Console.WriteLine($"B: {msg}");
}

3) Event với Action<T1,T2> (giống code bạn đang dùng)
using System;

public class CourseSimple
{
    private int _count;
    public event Action<int, int> OnNumberOfStudentChange; // old, new

    public void AddStudent()
    {
        int oldCount = _count;
        _count++;
        OnNumberOfStudentChange?.Invoke(oldCount, _count);
    }

    public void RemoveStudent()
    {
        if (_count == 0) return;
        int oldCount = _count;
        _count--;
        OnNumberOfStudentChange?.Invoke(oldCount, _count);
    }
}
